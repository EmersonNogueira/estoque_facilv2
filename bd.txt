CREATE TABLE estoques (
    id INT AUTO_INCREMENT PRIMARY KEY,           -- Código único para cada registro
    codigo_item INT NOT NULL,                     -- Referência para a tabela 'itens'
    codigo_local INT NOT NULL,                    -- Referência para a tabela 'locais'
    saldo INT NOT NULL DEFAULT 0,                 -- Quantidade de produtos
    validade DATE NULL,                           -- Pode ser NULL se a data não se aplicar
    FOREIGN KEY (codigo_item) REFERENCES itens(codigo_item) ON DELETE CASCADE,
    FOREIGN KEY (codigo_local) REFERENCES locais(codigo_local) ON DELETE CASCADE,
    UNIQUE KEY unique_estoques (codigo_item, codigo_local, validade)
);
